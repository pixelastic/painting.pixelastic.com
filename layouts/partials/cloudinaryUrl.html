{{- /* Generate Cloudinary URL for a Hugo image resource Parameters: - image: (resource) REQUIRED - Hugo resource object - transformations: (string) Optional Cloudinary transformations (default: "f_auto") Returns: Cloudinary URL in production, RelPermalink in development Usage: {{ partial "cloudinaryUrl.html" (dict "image" $imageResource) }} {{ partial "cloudinaryUrl.html" (dict "image" $imageResource "transformations" "f_auto,w_800") }} */ -}} {{- $image := .image -}} {{- $transformations := .transformations | default "f_auto" -}} {{- $url := "" -}} {{- if hugo.IsProduction -}} {{- $fingerprinted := $image | fingerprint "md5" -}} {{- $baseUrl := $fingerprinted.Permalink -}} {{- $url = printf "https://res.cloudinary.com/pixelastic-painting/image/fetch/%s/%s" $transformations $baseUrl -}} {{- else -}} {{- $url = $image.RelPermalink -}} {{- end -}} {{- $url -}}
