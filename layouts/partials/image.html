{{- /* Centralized partial for rendering images with automatic dimensions Parameters: - image: (resource|string) REQUIRED - Either a Hugo resource object OR a string (src/path) - alt: (string) Alt text - class: (string) CSS classes to apply - loading: (string) "lazy" | "eager" | "" (default: "lazy") Usage examples: With resource: {{ partial "image.html" (dict "image" $imageResource "alt" "My image") }} With string: {{ partial "image.html" (dict "image" "/path/to/image.png" "alt" "My image") }} No lazy load: {{ partial "image.html" (dict "image" $img "alt" "Hero" "loading" "eager") }} */ -}} {{- $image := .image -}} {{- $alt := .alt | default "" -}} {{- $class := .class | default "" -}} {{- $loading := .loading | default "lazy" -}} {{- $src := $image -}} {{- /* If image is a resource, we get its path */ -}} {{- $imageType := printf "%T" $image -}} {{- $imageIsResource := ne $imageType "string" -}} {{- if $imageIsResource -}} {{- /* In production, fingerprint the image for cache busting */ -}} {{- if hugo.IsProduction -}} {{- $fingerprinted := $image | fingerprint "md5" -}} {{- $src = $fingerprinted.RelPermalink -}} {{- else -}} {{- $src = $image.RelPermalink -}} {{- end -}} {{- end -}}
