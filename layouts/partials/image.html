{{- /* Centralized partial for rendering images with automatic dimensions Parameters: - image: (resource|string) REQUIRED - Either a Hugo resource object OR a string (src/path) - alt: (string) Alt text - class: (string) CSS classes to apply Usage examples: With resource: {{ partial "image.html" (dict "image" $imageResource "alt" "My image") }} With string: {{ partial "image.html" (dict "image" "/path/to/image.png" "alt" "My image") }} */ -}} {{- $image := .image -}} {{- $alt := .alt | default "" -}} {{- $class := .class | default "" -}} {{- $src := $image -}} {{- /* If image is a resource, we get its path */ -}} {{- $imageType := printf "%T" $image -}} {{- $imageIsResource := ne $imageType "string" -}} {{- if $imageIsResource -}} {{- $src = $image.RelPermalink -}} {{- end -}}
