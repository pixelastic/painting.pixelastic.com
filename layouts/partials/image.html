{{- /* Centralized partial for rendering images with automatic dimensions Parameters: - image: (resource|string) REQUIRED - Either a Hugo resource object OR a string (src/path) - alt: (string) Alt text - class: (string) CSS classes to apply - lazy: (bool) true = show only LQIP placeholder, false = load image directly (default: true) Usage examples: With resource: {{ partial "image.html" (dict "image" $imageResource "alt" "My image") }} With string: {{ partial "image.html" (dict "image" "/path/to/image.png" "alt" "My image") }} No lazy load: {{ partial "image.html" (dict "image" $img "alt" "Hero" "lazy" false) }} */ -}} {{- $image := .image -}} {{- $alt := .alt | default "" -}} {{- $class := .class | default "" -}} {{- $lazy := .lazy | default true -}} {{- $src := "" -}} {{- $lqipStyle := "" -}} {{- $lazyClass := "" -}} {{- if $lazy -}} {{- $lazyClass = "lazy lazy-loading" -}} {{- end -}} {{- /* If image is a resource, we get its path */ -}} {{- $imageType := printf "%T" $image -}} {{- $imageIsResource := ne $imageType "string" -}} {{- if $imageIsResource -}} {{- /* Generate LQIP for placeholder */ -}} {{- $lqip := $image.Resize "20x" -}} {{- $lqipBase64 := printf "data:image/jpeg;base64,%s" ($lqip.Content | base64Encode) -}} {{- $lqipStyle = printf "background-image: url('%s');" $lqipBase64 -}} {{- /* Get the final image URL */ -}} {{- if hugo.IsProduction -}} {{- $fingerprinted := $image | fingerprint "md5" -}} {{- $src = $fingerprinted.Permalink -}} {{- /* Proxy through Cloudinary for optimization */ -}} {{- $src = printf "https://res.cloudinary.com/pixelastic-painting/image/fetch/f_auto/%s" $src -}} {{- else -}} {{- $src = $image.RelPermalink -}} {{- end -}} {{- else -}} {{- /* String path, use as-is */ -}} {{- $src = $image -}} {{- end -}}
