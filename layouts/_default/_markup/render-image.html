{{- /* Try to find image as a page resource */ -}} {{- $destination := .Destination -}} {{- /* Remove ./ prefix if present to help GetMatch */ -}} {{- $destination = strings.TrimPrefix "./" $destination -}} {{- $image := .Page.Resources.GetMatch $destination -}} {{- if not $image -}} {{- /* Fallback to path string if not found as resource */ -}} {{- $image = .Destination -}} {{- end -}} {{- /* Get the URL for the lightbox link */ -}} {{- $imageUrl := "" -}} {{- $imageType := printf "%T" $image -}} {{- if ne $imageType "string" -}} {{- $imageUrl = partial "cloudinaryUrl.html" (dict "image" $image) -}} {{- else -}} {{- $imageUrl = $image -}} {{- end -}} {{- /* Count images for lazy loading logic */ -}} {{- $scratchKey := printf "imageCount-%s" .Page.RelPermalink -}} {{- $imageCount := .Page.Store.Get $scratchKey | default 0 -}} {{- .Page.Store.Set $scratchKey (add $imageCount 1) -}} {{- $lazy := true -}} {{- if le $imageCount 1 -}} {{- /* First 2 images (index 0 and 1) are not lazy */ -}} {{- $lazy = false -}} {{- end -}} {{- partial "image.html" (dict "image" $image "alt" .Text "lazy" $lazy ) -}}
